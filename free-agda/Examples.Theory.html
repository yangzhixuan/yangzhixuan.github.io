\begin{code}
module Examples.Theory where

open import Prelude
open import Signatures
open import FreeMonad.Theory hiding (Law; Theory)

record Law (𝔽 : Signature) : Type₁
record Theory (𝔽 : Signature) : Type₁

\end{code}
%<*law>
\begin{code}
record Law 𝔽 where
  field  Γ  : Type
         ν  : Γ → Type
         eqn : (γ : Γ) → Equation 𝔽 (ν γ)
\end{code}
%</law>
%<*theory>
\begin{code}
record Theory 𝔽 where
  field  Laws  : Type
         Eqns  : Laws → Law 𝔽
\end{code}
%</theory>
