module Examples.State where

open import Prelude
open import Signatures

data Op (S : Type) : Type 
\end{code}
%<*op>
\begin{code}
data Op S where
  `get  : Op S
  `put  : S → Op S
\end{code}
%</op>
\begin{code}
module _ {S : Type} where
\end{code}
%<*arity>
\begin{code}
  Arity : Op S → Type
  Arity `get      = S
  Arity (`put _)  = ⊤
\end{code}
%</arity>
\begin{code}
  𝕊 : Signature
\end{code}
%<*sig-inst>
\begin{code}
  𝕊 = Op S ◁ Arity
\end{code}
%</sig-inst>
