--------------------------------------------------------------------------------
-- Equations and theories for computations
--------------------------------------------------------------------------------

module FreeMonad.Theory where

open import FreeMonad.Syntax
private variable 𝔽 : Signature

--------------------------------------------------------------------------------
-- An equation
--------------------------------------------------------------------------------


infix 4 _≐_
record Equation (𝔽 : Signature) (ν : Type a) : Type a
\end{code}
%<*equation>
\begin{code}
record Equation 𝔽 ν where
  constructor _≐_
  field lhs rhs : Syntax 𝔽 ν
\end{code}
%</equation>
\begin{code}
open Equation public

∀ⁿ : ∀ {n} → (∀ {A : Type} → Curryⁿ n (Syntax 𝔽 A) (const (Equation 𝔽 A))) 
   → Equation 𝔽 (Fin n)
∀ⁿ {n = n} f = uncurryⁿ {A = Syntax _ (Fin n)} f var

--------------------------------------------------------------------------------
-- A law (equation with the type params as existentials)
--------------------------------------------------------------------------------

record Law (𝔽 : Signature) (ℓ : Level) : Type (ℓsuc ℓ)
record Law 𝔽 ℓ where
  field  Γ  : Type ℓ
         ν  : Γ → Type ℓ
         eqn : (γ : Γ) → Equation 𝔽 (ν γ)

--------------------------------------------------------------------------------
-- A theory
--------------------------------------------------------------------------------

record Theory (𝔽 : Signature) (ℓ : Level) : Type (ℓsuc ℓ)
record Theory 𝔽 ℓ where
  field  Laws  : Type
         Eqns  : Laws → Law 𝔽 ℓ

open Theory

_◁≡_ : {ℓ : Level} → (L : Type) → (L → Law 𝔽 ℓ) → Theory 𝔽 ℓ
(laws ◁≡ eqns) .Laws = laws
(laws ◁≡ eqns) .Eqns = eqns

open Law public

-----------------------------------------------------------------------------------
-- Algebras can "respect" an equation or a full theory
--------------------------------------------------------------------------------

module _ {𝔽 : Signature} {𝒞 : Type a} {ν : Type b} where
  private module RespectsExt where
    infix 3 _Respects_
    _Respects_ : 𝔽 -Alg[ 𝒞 ] → Equation 𝔽 ν → Type _
    ϕ Respects lhs ≐ rhs = flip (interp 𝔽 ϕ) lhs ≡ flip (interp 𝔽 ϕ) rhs

  infix 3 _Respects_
  _Respects_ : 𝔽 -Alg[ 𝒞 ] → Equation 𝔽 ν → Type _
\end{code}
%<*respects>
\begin{code}
  ϕ Respects (lhs ≐ rhs) = ∀ (ρ : ν → 𝒞) → interp 𝔽 ϕ ρ lhs ≡ interp 𝔽 ϕ ρ rhs
\end{code}
%</respects>
\begin{code}
infix 4 _Models_
_Models_ : 𝔽 -Alg[ A ] → Theory 𝔽 b → Type _
\end{code}
%<*in-theory>
\begin{code}
ϕ Models 𝒯 = ∀ i γ → ϕ Respects 𝒯 .Eqns i .eqn γ
\end{code}
%</in-theory>
\begin{code}

module _ {𝔽 : Signature} where
  _-Model[_] : Theory 𝔽 a → Type b → Type _
\end{code}
%<*t-model>
\begin{code}
  𝒯 -Model[ 𝒞 ] = Σ[ ϕ ⦂ 𝔽 -Alg[ 𝒞 ] ] × ϕ Models 𝒯
\end{code}
%</t-model>
\begin{code}

--------------------------------------------------------------------------------
-- A theory needs to have finitely many variables
--------------------------------------------------------------------------------

FiniteVars : Theory 𝔽 a → _
FiniteVars 𝒯 =
  ∀ i →
  let law = 𝒯 .Eqns i in
  (γ : Γ law) →
  SplitQuotientedChoiceω (ν law γ)

--------------------------------------------------------------------------------
-- In some cases we need variables to have decidable equality
--------------------------------------------------------------------------------
DiscreteVars : Theory 𝔽 a → _
DiscreteVars 𝒯 =
  ∀ i →
  let law = 𝒯 .Eqns i in
  (γ : Γ law) →
  Discrete (ν law γ)
\end{code}
